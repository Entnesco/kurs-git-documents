#basic comands
ls - list - lista elementów w folderze
ls -a - lista wszystkich elementów w folderze razem z ukrytymi
cd - zmina scieżki
cd .. - wycowanie do poziomu wyżej
start . - otwarcie obecnej lokalizacji
pwd - wyświetlenie obecnej ścieżki
touch - tworzenie plików
mkdir - tworzenie folderów
rm <nazwa> - usuwanie plików
rm -rf <nazwa> - usuwanie folderu
code . - otworzenie visual studio

#git commands
git init - inicjalizacja repo
git status - status repo
git add <nazwa> - dodanie do repo
git add . - dodanie do repo wszystkich plików
git commit -m "example message" - zacomitowanie dodanych zmian w repo
git log - log zmian na repo
git log --oneline - log jednolinijkowy
git commit --amend - zacomitowanie nowego elementu do poprzedniego commita
git branch - lista gałęzi 
git branch <branch_name> - tworzenie nowej gałęzi
git switch <branch_name> - zmiana gałęzi
git commit -a -m <comment> - git add. oraz git commit -m w jednym
git checkout <branch_name> - git switch and more
git switch -c <branch_name> - create and switch to new branch
git branch -D <branch_name> - force delete branch
git branch -m <new_name> - rename branch
git merge <branch_name> - merging branches
git branch -v - more info abaut evry branch
git diff - lista niezatwierdzonych zmian
git diff HEAD - lista wszystkich różnic
git diff --staged - lista różnic w dodanych plikach
git diff --cached - lista różnic w dodanych plikach
