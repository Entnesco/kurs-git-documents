###basic comands###
ls - list - lista elementów w folderze
ls -a - lista wszystkich elementów w folderze razem z ukrytymi
cd - zmina scieżki
cd .. - wycowanie do poziomu wyżej
start . - otwarcie obecnej lokalizacji
pwd - wyświetlenie obecnej ścieżki
touch - tworzenie plików
mkdir - tworzenie folderów
rm <nazwa> - usuwanie plików
rm -rf <nazwa> - usuwanie folderu
code . - otworzenie visual studio


###git commands###

#basic comands
git init - inicjalizacja repo
git status - status repo
git add <nazwa> - dodanie do repo
git add . - dodanie do repo wszystkich plików
git commit -m "example message" - zacomitowanie dodanych zmian w repo
git log - log zmian na repo
git log --oneline - log jednolinijkowy
git commit --amend - zacomitowanie nowego elementu do poprzedniego commita

#git branch commands
git branch - lista gałęzi 
git branch <branch_name> - tworzenie nowej gałęzi
git switch <branch_name> - zmiana gałęzi
git commit -a -m <comment> - git add. oraz git commit -m w jednym
git checkout <branch_name> - git switch and more
git switch -c <branch_name> - create and switch to new branch
git branch -D <branch_name> - force delete branch
git branch -m <new_name> - rename branch
git branch -v - more info abaut evry branch

#git merge command
git merge <branch_name> - merging branches

#git diff command
git diff - lista niezatwierdzonych zmian
git diff HEAD - lista wszystkich różnic
git diff --staged - lista różnic w dodanych plikach
git diff --cached - lista różnic w dodanych plikach
git diff HEAD <filename> - porównanie z wskazanym plikiem
git diff --cached <filename> - porównanie z wskazanym plikiem
git diff --staged <filename> - porównanie z wskazanym plikiem
git diff branch1 branch2 - porówniania 2 branchów
git diff commit1 commit2 - porównania 2 commitów

#git stash command
git stash - zapisanie niezacomitowanych zmian
git stash pop - uwolnienie zapisanych zmian
git stash apply - wklejenie zapisanych zmian bez ich uwalniania
git stash list - lista zapisanych zmian
git stash apply stash@{2} - wklejenie konkretnej zmiany z listy zmian
git stash drop stash@{2} - usuwanie konkretniej zmiany z listy zmian
git stash clear - usuwanie wszystkich zmian z listy zmian

#git checkout command
git checkout <commit_hash> - cofanie do wybranego commita
#if u wanna go back to current branch use "git switch"
#u can create new branch if u go back to previous commit
git checkout HEAD~1 - go back to parant
git checkout HEAD~2 - go back to grandparent~
git switch - - go back to previouse branch
git checkout HEAD <file> -to discard any changes in that file
git checkout -- <file> -to discard any changes in that file

#git restore command
git restore <file-name> - discard any changes in that file
git restore --source HEAD~1 <file> - restore file form previouse commits
git restore --staged <file> -unstage staged files
